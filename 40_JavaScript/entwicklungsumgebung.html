JavaScript in Webseite einbinden.


<!DOCTYPE html>
<html>

<body>
<script type="application/javascript">

		function addiere(summand1, summand2) {
				return summand1, summand2;
		}

		var summe = addiere(3,5);

</script>
</body>

</html>

Ausgabe mit JavaScript


<!DOCTYPE html>
<html>

<body>
<script type="application/javascript">

		function addiere(summand1, summand2) {
				return summand1, summand2;
		}

		var summe = addiere(3,5);

	// ins die HTML-Datei schreiben
	document.write(summe);

	// in die console schreiben
	// https://getfirebug.com/logging
	console.log(summe);

</script>
</body>

</html>
